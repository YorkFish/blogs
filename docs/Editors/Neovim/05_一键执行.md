# 一键执行

设置 <kbd>m</kbd> 为快捷键

    noremap m :call CompileRunGcc()<CR>
    func! CompileRunGcc()
        exec "w"
        if &filetype == 'c'
            exec "!gcc -std=c11 % -o %<"
            exec "!%<"
        elseif &filetype == 'cpp' || &filetype == 'cc'
            exec "!g++ -Wall -std=c++14 % -o %<"
            exec "!%<"
        elseif &filetype == 'python'
            :sp
            :term python %
        elseif &filetype == 'markdown'
            exec "MarkdownPreview"
        endif
    endfunc
